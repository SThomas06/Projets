1   package controller;

3   import java.awt.event.KeyEvent;
4   
5   public class EventPerformer {
6   	private ControllerFacade orderPerformer;
7   	
8   	public EventPerformer(ControllerFacade orderPerformer) {
9   		this.orderPerformer = orderPerformer;
10  	}
11  	
12  	public void eventPerform(KeyEvent keyEvent) {
13  		this.orderPerformer.orderPerform(this.keyCodeToOrder(keyEvent.getKeyCode()));
14  	}
15  	
16  	private Order keyCodeToOrder(int keyCode) {
17  		switch(keyCode) {
18  		case KeyEvent.VK_UP:
19  			return Order.UP;
20  		case KeyEvent.VK_LEFT:
21  			return Order.LEFT;
22  		case KeyEvent.VK_DOWN:
23  			return Order.DOWN;
24  		case KeyEvent.VK_RIGHT:
25  			return Order.RIGHT;
26  		case KeyEvent.VK_SPACE:
27  			return Order.SPELL;
28  		case KeyEvent.VK_ESCAPE:
29  			return Order.ESCAPE;
30  		default:
31  			return Order.NOPE;
32  		}
33  	}
34  }
